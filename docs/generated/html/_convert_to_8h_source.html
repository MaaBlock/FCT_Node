152 template<typename ReturnType, typename... Args>
153 inline v8::Local<v8::Value> convertToJS(NodeEnvironment& env,
154 std::function<ReturnType(Args...)> func) {
155 if (!func) {
156 return v8::Null(env.isolate());
157 }
158
159 auto wrapper = std::make_unique<FunctionWrapper<ReturnType, Args...>>(std::move(func), env);
160 int64_t id = env.functionManager().registerFunction(std::move(wrapper));
161
162 v8::Local<v8::Array> callbackData = v8::Array::New(env.isolate(), 2);
163 callbackData->Set(env.context(), 0, v8::BigInt::New(env.isolate(), id)).Check();
164 callbackData->Set(env.context(), 1, v8::External::New(env.isolate(), &env)).Check();
165
166 v8::Local<v8::FunctionTemplate> funcTemplate = v8::FunctionTemplate::New(
167 env.isolate(), GlobalFunctionCallback, callbackData);
168
169 v8::Local<v8::Context> context = env.isolate()->GetCurrentContext();
170 v8::Local<v8::Function> jsFunc = funcTemplate->GetFunction(context).ToLocalChecked();
171
172 return jsFunc;
173 }
