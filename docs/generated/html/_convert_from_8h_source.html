198 v8::HandleScope handle_scope(isolate);

201
202 v8::Local<v8::Function> func = v8::Local<v8::Function>::New(isolate, persistentFunc);
203
204 std::vector<v8::Local<v8::Value>> jsArgs;
205 if constexpr (sizeof...(args) > 0) {
206 jsArgs.reserve(sizeof...(args));
207 (jsArgs.push_back(convertToJS(isolate, args)), ...);
208 }
209
210 v8::TryCatch try_catch(isolate);

212 jsArgs.size(), jsArgs.data());

217 throw std::runtime_error("JavaScript function error: " + std::string(*exception_str));
218 }
